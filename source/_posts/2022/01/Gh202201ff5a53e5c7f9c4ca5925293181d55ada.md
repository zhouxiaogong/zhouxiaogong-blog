---
title: 151. 翻转字符串里的单词
date: 2022-01-06 22:35:03
---

```
class Solution {
public:
    string reverseWords(string s) {
        vector<string> ans;
        int size=s.size();
        int i=0;
        int j=size-1;
        while(s[i]==' ')i++;
        while(s[j]==' ')j--;//去除开头和结尾的空格
        int k=i;
        int h=i;
        while(k<=j)
        {
            while(k<=j&&s[k]!=' ')k++;
            string temp=s.substr(h,k-h);
            ans.push_back(temp);
            while(s[k]==' ')k++;//去除中间的空格
            h=k;
        }
        int anssize=ans.size();
        string res="";
        for(int g=anssize-1;g>=0;g--)
        {
            res+=ans[g];
            if(g!=0)res+=' ';
        }
        return res;
    }
};
```
